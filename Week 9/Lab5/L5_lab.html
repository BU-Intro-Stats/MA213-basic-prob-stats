Let’s discuss this statement with your group.

Based on a 2025 Research poll on 1,500 adults: “We are 95% confident that 60% to 64% of Americans think the federal government does not do enough for middle class people.

Population given

#
pop <- c(rep("YouTube", 10000), rep("No YouTube", 4000))
n <- length(pop)
true_prop = sum(pop == "YouTube") / n

Sampling the interval

# 95% CI 

n <- 100
K <- 100


sample_data <- data.frame(
  Sample_number = integer(0),
  Proportion_est = numeric(0),
  Lower = numeric(0),
  Upper = numeric(0)
)

for (k in 1:K) {
  my_sample <- sample(pop, n, replace = TRUE)
  p_hat <- sum(my_sample == "YouTube") / n
  SE <- sqrt((p_hat * (1 - p_hat)) / n)
  lower <- p_hat - 1.96 * SE
  upper <- p_hat + 1.96 * SE
  sample_data <- rbind(sample_data, data.frame(
    Sample_number = k,
    Proportion_est = p_hat,
    Lower = lower,
    Upper = upper
  ))
}

How many intervals include the population parameter value?

#
#
#
CI_95 <- sample_data %>% mutate(Capture = ifelse(Lower < true_prop, ifelse(Upper > true_prop, 1, 0), 0))

sum(CI_95$Capture)/length(CI_95$Capture)
## [1] 0.94

Plot to explain

CI_plot_95 <- CI_95 %>% ggplot(aes(x = Sample_number, y = Proportion_est)) + geom_point(aes(color = Capture)) + geom_errorbar(aes(ymin = Lower, ymax = Upper, color = Capture)) + coord_flip() + geom_hline(yintercept = true_prop, linetype = "dashed", color = "blue") + labs(title = "95% Confidence Intervals") + theme(plot.title = element_text(hjust = 0.5)) + ylim(true_prop - 0.2, true_prop + 0.2)
